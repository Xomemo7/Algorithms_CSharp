static List<int> selectionSort(List<int>list)
 {
     int max;
     int maxIndex;
     
     int unsortedListLength=list.Count;
     while (unsortedListLength > 1)
     {
         max = list[0];
         maxIndex=0;
         for (int i = 0; i < unsortedListLength; i++)
         {
             if (max < list[i])
             {
                 max = list[i];
                 maxIndex = i;
             }

         }
         Swap(ref list, unsortedListLength - 1, maxIndex);
         
         unsortedListLength--;
     }
     

     return list;


 }
